```c++
#include <bits/stdc++.h>
#define endl '\n'
#define INF 987654321
#define p_q priority_queue
#define pbk push_back
#define rep(i, a, b) for (int i=a; i<=b; i++) 
#define all(v) (v).begin(), (v).end()


using namespace std;
int T, N, K, M, S, H, W, Q;
#define MAX 501
int matrix[MAX][MAX] {};
int dp[MAX][MAX] {};
void Solve() {
    cin >> N >> M;
    rep(i,1,N) {
        rep(j,1,M) {
            cin >> matrix[i][j];
        }
    }

    rep(i,1,N) {
        rep(j,1,M) {
            dp[i][j] = INF;
            rep(k,1,M) {
                if(j==k) continue;

                dp[i][j] = min(dp[i][j], dp[i-1][k]+matrix[i][j]);
            }
        }
    }
    
    cout << *min_element(dp[N]+1, dp[N]+M+1);
}
int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);
    Solve();
}
```
