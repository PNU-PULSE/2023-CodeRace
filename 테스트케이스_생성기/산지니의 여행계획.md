### 산지니의 여행계획

```python
import random

file = open('generate_random.inp', 'w')
N = random.randrange(5, 50000)
M = random.randrange(N-1, 1000000)
file.write(str(N) + ' ' + str(M) + '\n')
for i in range(1, N):
    file.write(str(i) + ' ' + str(i+1) + ' ' + str(random.randrange(1,10000)) + '\n')

for _ in range(M-(N-1)):
    a = random.randrange(1, N+1)
    b = random.randrange(1, N+1)
    c = random.randrange(1, 10000)
    while(a==b) :
        b = random.randrange(1,N+1)
    
    file.write(str(a) + ' ' + str(b) + ' ' + str(c) + '\n')
```
