int getParent(int num) {
    if (parent[num] == num) return num;
    return parent[num] = getParent(parent[num]);
}
void unionParent(int a, int b) {
    a = getParent(a);
    b = getParent(b);
    if (a < b) parent[b] = a;
    else parent[a] = b;
}

bool isSameParent(int a, int b) {
    return getParent(a) == getParent(b);
}

class Edge {
public:
    int node[2];
    int dis;

        node[0] = a;
        node[1] = b;
        dis = _dis;
    }
    bool operator < (const Edge& edge) {
        return dis < edge.dis;
    }

void Kruskal() { //MST
    cin >> N >> M; //N is node, M is edge, Node starts from 1 to N
    for (int i = 0; i < M; i++) {
        int s, e, dis;

            kruskalEdge[nodeA].pbk({nodeB, dis});

    }
}

int dp[MAX] {};
void dfs(int node) {
    visited[node]=1;
    for(auto next : kruskalEdge[node]) {
        int nextNode = next.first;
        int nextDis = next.second;
        if(visited[nextNode]) continue;
        dfs(nextNode);
        dp[node] += dp[nextNode] + nextDis;
    }
}

void dfsSumLeaf(int node) {
    visited[node] = 1;
    for(auto next : kruskalEdge[node]) {
        int nextNode = next.first;
        int nextDis = next.second;
        if(visited[nextNode]) continue;
        dp[nextNode] += nextDis + dp[node];
        dfsSumLeaf(nextNode);
    }
}

    memset(dp, 0, sizeof(dp));
    dfsSumLeaf(S);
    
    int lastTraverse = *max_element(dp+1, dp+N+1);
    answer -= lastTraverse;
    cout << answer;
}

    ios::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);
    Solve();
}
